# ThisM Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a lightweight self-hosted server monitoring tool with a Go backend, WebSocket-based agents, SQLite storage, and a React/Tailwind/shadcn/ui frontend.

**Architecture:** A single `thisM-server` binary (Go) embeds the compiled React frontend and exposes HTTP REST + WebSocket endpoints. A separate `thisM-agent` binary runs on each monitored server, connects via WebSocket long-connection, and pushes metrics. SQLite stores all historical data.

**Tech Stack:** Go 1.22+, chi router, gorilla/websocket, modernc.org/sqlite (pure Go, no CGo), gopsutil v3, React 18 + TypeScript, Vite, Tailwind CSS v3, shadcn/ui, Recharts.

```go
	IP        string `json:"ip" db:"ip"`

// MetricsPayload is the JSON message sent by agents over WebSocket.
type MetricsPayload struct {
	Type      string      `json:"type"`
	TS        int64       `json:"ts"`
	CPU       float64     `json:"cpu"`
	Mem       MemStats    `json:"mem"`
	Disk      []DiskStats `json:"disk"`
	Net       NetStats    `json:"net"`
	Processes []Process   `json:"processes"`
	Services  []Service   `json:"services"`
}

type MemStats struct {
	Used  uint64 `json:"used"`
	Total uint64 `json:"total"`
}

type DiskStats struct {
	Mount string `json:"mount"`
	Used  uint64 `json:"used"`
	Total uint64 `json:"total"`
}

type NetStats struct {
	RxBytes uint64 `json:"rx_bytes"`
	TxBytes uint64 `json:"tx_bytes"`
}

type Process struct {
	PID        int32   `json:"pid"`
	Name       string  `json:"name"`
	CPUPercent float64 `json:"cpu"`
	MemRSS     uint64  `json:"mem"`
}

type Service struct {
	Name   string `json:"name"`
	Status string `json:"status"`
}

// WSMessage wraps any WebSocket message with a type discriminator.
type WSMessage struct {
	Type    string      `json:"type"`
	Payload interface{} `json:"payload,omitempty"`
}
```

```go
package store

import (
	"database/sql"
	"fmt"
	"time"

	"github.com/yourusername/thism/internal/models"
	_ "modernc.org/sqlite"
)

type Store struct {
	db *sql.DB
}

func New(path string) (*Store, error) {
	db, err := sql.Open("sqlite", path)
	if err != nil {
		return nil, fmt.Errorf("open db: %w", err)
	}
	db.SetMaxOpenConns(1) // SQLite is single-writer
	s := &Store{db: db}
	if err := s.migrate(); err != nil {
		return nil, fmt.Errorf("migrate: %w", err)
	}
	return s, nil
}

func (s *Store) Close() error { return s.db.Close() }

func (s *Store) migrate() error {
	_, err := s.db.Exec(`
		CREATE TABLE IF NOT EXISTS nodes (
			id         TEXT PRIMARY KEY,
			name       TEXT NOT NULL,
			token      TEXT NOT NULL UNIQUE,
			ip         TEXT DEFAULT '',
			os         TEXT DEFAULT '',
			arch       TEXT DEFAULT '',
			created_at INTEGER NOT NULL,
			last_seen  INTEGER NOT NULL
		);
		CREATE TABLE IF NOT EXISTS metrics (
			id          INTEGER PRIMARY KEY AUTOINCREMENT,
			node_id     TEXT NOT NULL,
			ts          INTEGER NOT NULL,
			cpu_percent REAL DEFAULT 0,
			mem_used    INTEGER DEFAULT 0,
			mem_total   INTEGER DEFAULT 0,
			disk_used   INTEGER DEFAULT 0,
			disk_total  INTEGER DEFAULT 0,
			net_rx      INTEGER DEFAULT 0,
			net_tx      INTEGER DEFAULT 0
		);
		CREATE INDEX IF NOT EXISTS idx_metrics_node_ts ON metrics(node_id, ts);
		CREATE TABLE IF NOT EXISTS processes (
			id          INTEGER PRIMARY KEY AUTOINCREMENT,
			node_id     TEXT NOT NULL UNIQUE,
			ts          INTEGER NOT NULL,
			data        TEXT NOT NULL
		);
		CREATE TABLE IF NOT EXISTS service_checks (
			id           INTEGER PRIMARY KEY AUTOINCREMENT,
			node_id      TEXT NOT NULL,
			name         TEXT NOT NULL,
			status       TEXT DEFAULT 'unknown',
			last_checked INTEGER NOT NULL,
			UNIQUE(node_id, name)
		);
	`)
	return err
}

func (s *Store) UpsertNode(n *models.Node) error {
	_, err := s.db.Exec(`
		INSERT INTO nodes (id, name, token, ip, os, arch, created_at, last_seen)
		VALUES (?, ?, ?, ?, ?, ?, ?, ?)
		ON CONFLICT(id) DO UPDATE SET
			name=excluded.name, ip=excluded.ip, os=excluded.os,
			arch=excluded.arch, last_seen=excluded.last_seen
	`, n.ID, n.Name, n.Token, n.IP, n.OS, n.Arch, n.CreatedAt, n.LastSeen)
	return err
}

func (s *Store) GetNodeByToken(token string) (*models.Node, error) {
	row := s.db.QueryRow(`SELECT id, name, token, ip, os, arch, created_at, last_seen FROM nodes WHERE token = ?`, token)
	var n models.Node
	err := row.Scan(&n.ID, &n.Name, &n.Token, &n.IP, &n.OS, &n.Arch, &n.CreatedAt, &n.LastSeen)
	if err == sql.ErrNoRows {
		return nil, nil
	}
	return &n, err
}

func (s *Store) ListNodes() ([]*models.Node, error) {
	rows, err := s.db.Query(`SELECT id, name, token, ip, os, arch, created_at, last_seen FROM nodes ORDER BY created_at`)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var nodes []*models.Node
	for rows.Next() {
		var n models.Node
		if err := rows.Scan(&n.ID, &n.Name, &n.Token, &n.IP, &n.OS, &n.Arch, &n.CreatedAt, &n.LastSeen); err != nil {
			return nil, err
		}
		nodes = append(nodes, &n)
	}
	return nodes, rows.Err()
}

func (s *Store) UpdateLastSeen(nodeID string) error {
	_, err := s.db.Exec(`UPDATE nodes SET last_seen = ? WHERE id = ?`, time.Now().Unix(), nodeID)
	return err
}

func (s *Store) InsertMetrics(nodeID string, m *models.MetricsPayload) error {
	diskUsed, diskTotal := aggregateDisk(m.Disk)
	_, err := s.db.Exec(`
		INSERT INTO metrics (node_id, ts, cpu_percent, mem_used, mem_total, disk_used, disk_total, net_rx, net_tx)
		VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
	`, nodeID, m.TS, m.CPU, m.Mem.Used, m.Mem.Total, diskUsed, diskTotal, m.Net.RxBytes, m.Net.TxBytes)
	return err
}

func aggregateDisk(disks []models.DiskStats) (used, total uint64) {
	for _, d := range disks {
		used += d.Used
		total += d.Total
	}
	return
}

// MetricsRow is a flattened metrics record for API responses.
type MetricsRow struct {
	TS         int64   `json:"ts"`
	CPUPercent float64 `json:"cpu"`
	MemUsed    uint64  `json:"mem_used"`
	MemTotal   uint64  `json:"mem_total"`
	DiskUsed   uint64  `json:"disk_used"`
	DiskTotal  uint64  `json:"disk_total"`
	NetRx      uint64  `json:"net_rx"`
	NetTx      uint64  `json:"net_tx"`
}

func (s *Store) QueryMetrics(nodeID string, from, to int64) ([]*MetricsRow, error) {
	rows, err := s.db.Query(`
		SELECT ts, cpu_percent, mem_used, mem_total, disk_used, disk_total, net_rx, net_tx
		FROM metrics WHERE node_id = ? AND ts BETWEEN ? AND ? ORDER BY ts
	`, nodeID, from, to)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var result []*MetricsRow
	for rows.Next() {
		var r MetricsRow
		if err := rows.Scan(&r.TS, &r.CPUPercent, &r.MemUsed, &r.MemTotal, &r.DiskUsed, &r.DiskTotal, &r.NetRx, &r.NetTx); err != nil {
			return nil, err
		}
		result = append(result, &r)
	}
	return result, rows.Err()
}

func (s *Store) UpsertProcesses(nodeID string, ts int64, data string) error {
	_, err := s.db.Exec(`
		INSERT INTO processes (node_id, ts, data) VALUES (?, ?, ?)
		ON CONFLICT(node_id) DO UPDATE SET ts=excluded.ts, data=excluded.data
	`, nodeID, ts, data)
	return err
}

func (s *Store) GetProcesses(nodeID string) (string, error) {
	var data string
	err := s.db.QueryRow(`SELECT data FROM processes WHERE node_id = ?`, nodeID).Scan(&data)
	if err == sql.ErrNoRows {
		return "[]", nil
	}
	return data, err
}

func (s *Store) UpsertServiceCheck(nodeID, name, status string) error {
	_, err := s.db.Exec(`
		INSERT INTO service_checks (node_id, name, status, last_checked) VALUES (?, ?, ?, ?)
		ON CONFLICT(node_id, name) DO UPDATE SET status=excluded.status, last_checked=excluded.last_checked
	`, nodeID, name, status, time.Now().Unix())
	return err
}

func (s *Store) GetServiceChecks(nodeID string) ([]map[string]interface{}, error) {
	rows, err := s.db.Query(`SELECT name, status, last_checked FROM service_checks WHERE node_id = ?`, nodeID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var result []map[string]interface{}
	for rows.Next() {
		var name, status string
		var lastChecked int64
		if err := rows.Scan(&name, &status, &lastChecked); err != nil {
			return nil, err
		}
		result = append(result, map[string]interface{}{"name": name, "status": status, "last_checked": lastChecked})
	}
	return result, rows.Err()
}

// PruneOldMetrics deletes metrics older than retentionDays.
func (s *Store) PruneOldMetrics(retentionDays int) error {
	cutoff := time.Now().AddDate(0, 0, -retentionDays).Unix()
	_, err := s.db.Exec(`DELETE FROM metrics WHERE ts < ?`, cutoff)
	return err
}
```

Create `internal/hub/hub.go`:

```go
package hub

import (
	"sync"

	"github.com/gorilla/websocket"
	"github.com/yourusername/thism/internal/models"
	"github.com/yourusername/thism/internal/store"
)

type agentConn struct {
	nodeID string
	conn   *websocket.Conn
	send   chan []byte
}

type Hub struct {
	store       *store.Store
	agents      map[string]*agentConn // nodeID → conn
	subscribers []chan models.WSMessage
	mu          sync.RWMutex
	register    chan *agentConn
	unregister  chan string
	broadcast   chan models.WSMessage
}

func New(s *store.Store) *Hub {
	return &Hub{
		store:      s,
		agents:     make(map[string]*agentConn),
		register:   make(chan *agentConn, 16),
		unregister: make(chan string, 16),
		broadcast:  make(chan models.WSMessage, 64),
	}
}

func (h *Hub) Run() {
	for {
		select {
		case conn := <-h.register:
			h.mu.Lock()
			h.agents[conn.nodeID] = conn
			h.mu.Unlock()
			h.Broadcast(models.WSMessage{Type: "node_status", Payload: map[string]interface{}{
				"node_id": conn.nodeID,
				"online":  true,
			}})

		case nodeID := <-h.unregister:
			h.mu.Lock()
			if conn, ok := h.agents[nodeID]; ok {
				close(conn.send)
				delete(h.agents, nodeID)
			}
			h.mu.Unlock()
			h.Broadcast(models.WSMessage{Type: "node_status", Payload: map[string]interface{}{
				"node_id": nodeID,
				"online":  false,
			}})

		case msg := <-h.broadcast:
			h.mu.RLock()
			subs := make([]chan models.WSMessage, len(h.subscribers))
			copy(subs, h.subscribers)
			h.mu.RUnlock()
			for _, sub := range subs {
				select {
				case sub <- msg:
				default:
				}
			}
		}
	}
}

func (h *Hub) Register(nodeID string, conn *websocket.Conn) {
	ac := &agentConn{nodeID: nodeID, conn: conn, send: make(chan []byte, 32)}
	h.register <- ac
}

func (h *Hub) Unregister(nodeID string) {
	h.unregister <- nodeID
}

func (h *Hub) IsOnline(nodeID string) bool {
	h.mu.RLock()
	defer h.mu.RUnlock()
	_, ok := h.agents[nodeID]
	return ok
}

func (h *Hub) OnlineNodeIDs() []string {
	h.mu.RLock()
	defer h.mu.RUnlock()
	ids := make([]string, 0, len(h.agents))
	for id := range h.agents {
		ids = append(ids, id)
	}
	return ids
}

func (h *Hub) Broadcast(msg models.WSMessage) {
	h.broadcast <- msg
}

func (h *Hub) Subscribe(ch chan models.WSMessage) {
	h.mu.Lock()
	defer h.mu.Unlock()
	h.subscribers = append(h.subscribers, ch)
}

func (h *Hub) Unsubscribe(ch chan models.WSMessage) {
	h.mu.Lock()
	defer h.mu.Unlock()
	for i, sub := range h.subscribers {
		if sub == ch {
			h.subscribers = append(h.subscribers[:i], h.subscribers[i+1:]...)
			return
		}
	}
}
```

```go
package api

import (
	"encoding/json"
	"fmt"
	"io"
	"log"
	"net/http"
	"strconv"
	"strings"
	"time"

	"github.com/go-chi/chi/v5"
	"github.com/go-chi/chi/v5/middleware"
	"github.com/gorilla/websocket"
	"github.com/yourusername/thism/internal/hub"
	"github.com/yourusername/thism/internal/models"
	"github.com/yourusername/thism/internal/store"
)

var upgrader = websocket.Upgrader{
	CheckOrigin: func(r *http.Request) bool { return true },
}

type Router struct {
	store      *store.Store
	hub        *hub.Hub
	adminToken string
}

func NewRouter(s *store.Store, h *hub.Hub, adminToken string) http.Handler {
	r := &Router{store: s, hub: h, adminToken: adminToken}
	mux := chi.NewRouter()
	mux.Use(middleware.Logger)
	mux.Use(middleware.Recoverer)

	// Agent WebSocket (uses node token)
	mux.Get("/ws/agent", r.handleAgentWS)

	// Dashboard WebSocket (uses admin token)
	mux.Get("/ws/dashboard", r.authMiddleware(r.handleDashboardWS))

	// REST API (admin token required)
	mux.Route("/api", func(api chi.Router) {
		api.Use(r.authMiddleware)
		api.Get("/nodes", r.handleListNodes)
		api.Get("/nodes/{id}/metrics", r.handleGetMetrics)
		api.Get("/nodes/{id}/processes", r.handleGetProcesses)
		api.Get("/nodes/{id}/services", r.handleGetServices)
	})

	return mux
}

func (rt *Router) authMiddleware(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		token := strings.TrimPrefix(r.Header.Get("Authorization"), "Bearer ")
		if token == "" {
			token = r.URL.Query().Get("token")
		}
		if token != rt.adminToken {
			http.Error(w, "unauthorized", http.StatusUnauthorized)
			return
		}
		next.ServeHTTP(w, r)
	})
}

func jsonOK(w http.ResponseWriter, v interface{}) {
	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(v)
}

// handleAgentWS handles a new agent WebSocket connection.
func (rt *Router) handleAgentWS(w http.ResponseWriter, r *http.Request) {
	token := r.URL.Query().Get("token")
	node, err := rt.store.GetNodeByToken(token)
	if err != nil || node == nil {
		http.Error(w, "invalid token", http.StatusUnauthorized)
		return
	}

	conn, err := upgrader.Upgrade(w, r, nil)
	if err != nil {
		log.Printf("ws upgrade: %v", err)
		return
	}
	defer conn.Close()

	// Update last seen and mark online
	rt.store.UpdateLastSeen(node.ID)
	rt.hub.Register(node.ID, conn)
	defer rt.hub.Unregister(node.ID)

	log.Printf("agent connected: %s (%s)", node.Name, node.ID)

	for {
		_, msg, err := conn.ReadMessage()
		if err != nil {
			log.Printf("agent %s disconnected: %v", node.ID, err)
			return
		}
		var payload models.MetricsPayload
		if err := json.Unmarshal(msg, &payload); err != nil {
			continue
		}
		rt.store.UpdateLastSeen(node.ID)
		rt.store.InsertMetrics(node.ID, &payload)

		// Persist processes as JSON blob
		if len(payload.Processes) > 0 {
			b, _ := json.Marshal(payload.Processes)
			rt.store.UpsertProcesses(node.ID, payload.TS, string(b))
		}

		// Persist service checks
		for _, svc := range payload.Services {
			rt.store.UpsertServiceCheck(node.ID, svc.Name, svc.Status)
		}

		// Broadcast to dashboard subscribers
		rt.hub.Broadcast(models.WSMessage{
			Type:    "metrics",
			Payload: map[string]interface{}{"node_id": node.ID, "data": payload},
		})
	}
}

// handleDashboardWS sends real-time updates to the web UI.
func (rt *Router) handleDashboardWS(w http.ResponseWriter, r *http.Request) {
	conn, err := upgrader.Upgrade(w, r, nil)
	if err != nil {
		return
	}
	defer conn.Close()

	ch := make(chan models.WSMessage, 32)
	rt.hub.Subscribe(ch)
	defer rt.hub.Unsubscribe(ch)

	for msg := range ch {
		b, _ := json.Marshal(msg)
		if err := conn.WriteMessage(websocket.TextMessage, b); err != nil {
			return
		}
	}
}

func (rt *Router) handleListNodes(w http.ResponseWriter, r *http.Request) {
	nodes, err := rt.store.ListNodes()
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	onlineIDs := make(map[string]bool)
	for _, id := range rt.hub.OnlineNodeIDs() {
		onlineIDs[id] = true
	}
	for _, n := range nodes {
		n.Online = onlineIDs[n.ID]
	}
	if nodes == nil {
		nodes = []*models.Node{}
	}
	jsonOK(w, map[string]interface{}{"nodes": nodes})
}

func (rt *Router) handleGetMetrics(w http.ResponseWriter, r *http.Request) {
	nodeID := chi.URLParam(r, "id")
	now := time.Now().Unix()
	from := now - 3600 // default: last 1 hour
	to := now

	if v := r.URL.Query().Get("from"); v != "" {
		from, _ = strconv.ParseInt(v, 10, 64)
	}
	if v := r.URL.Query().Get("to"); v != "" {
		to, _ = strconv.ParseInt(v, 10, 64)
	}

	rows, err := rt.store.QueryMetrics(nodeID, from, to)
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	if rows == nil {
		rows = []*store.MetricsRow{}
	}
	jsonOK(w, map[string]interface{}{"metrics": rows})
}

func (rt *Router) handleGetProcesses(w http.ResponseWriter, r *http.Request) {
	nodeID := chi.URLParam(r, "id")
	data, err := rt.store.GetProcesses(nodeID)
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	fmt.Fprint(w, data) // already JSON
}

func (rt *Router) handleGetServices(w http.ResponseWriter, r *http.Request) {
	nodeID := chi.URLParam(r, "id")
	checks, err := rt.store.GetServiceChecks(nodeID)
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	if checks == nil {
		checks = []map[string]interface{}{}
	}
	jsonOK(w, map[string]interface{}{"services": checks})
}
```

```go
package collector

import (
	"encoding/json"
	"log"
	"math"
	"net/url"
	"time"

	"github.com/gorilla/websocket"
	"github.com/shirou/gopsutil/v3/cpu"
	"github.com/shirou/gopsutil/v3/disk"
	"github.com/shirou/gopsutil/v3/mem"
	"github.com/shirou/gopsutil/v3/net"
	"github.com/shirou/gopsutil/v3/process"
	"github.com/yourusername/thism/internal/models"
)

const collectInterval = 5 * time.Second

type Collector struct {
	serverURL string
	token     string
	name      string
}

func New(serverURL, token, name string) *Collector {
	return &Collector{serverURL: serverURL, token: token, name: name}
}

// Collect gathers a single snapshot of system metrics.
func (c *Collector) Collect() (*models.MetricsPayload, error) {
	cpuPct, err := cpu.Percent(time.Second, false)
	if err != nil {
		return nil, err
	}
	vmStat, err := mem.VirtualMemory()
	if err != nil {
		return nil, err
	}
	partitions, err := disk.Partitions(false)
	if err != nil {
		return nil, err
	}
	var diskStats []models.DiskStats
	for _, p := range partitions {
		usage, err := disk.Usage(p.Mountpoint)
		if err != nil {
			continue
		}
		diskStats = append(diskStats, models.DiskStats{
			Mount: p.Mountpoint,
			Used:  usage.Used,
			Total: usage.Total,
		})
	}

	netCounters, err := net.IOCounters(false)
	var netStat models.NetStats
	if err == nil && len(netCounters) > 0 {
		netStat.RxBytes = netCounters[0].BytesRecv
		netStat.TxBytes = netCounters[0].BytesSent
	}

	procs, _ := process.Processes()
	var topProcs []models.Process
	for _, p := range procs {
		name, _ := p.Name()
		cpuP, _ := p.CPUPercent()
		memInfo, _ := p.MemoryInfo()
		var rss uint64
		if memInfo != nil {
			rss = memInfo.RSS
		}
		topProcs = append(topProcs, models.Process{
			PID:        p.Pid,
			Name:       name,
			CPUPercent: cpuP,
			MemRSS:     rss,
		})
		if len(topProcs) >= 30 {
			break
		}
	}

	cpuVal := 0.0
	if len(cpuPct) > 0 {
		cpuVal = cpuPct[0]
	}

	return &models.MetricsPayload{
		Type:      "metrics",
		TS:        time.Now().Unix(),
		CPU:       cpuVal,
		Mem:       models.MemStats{Used: vmStat.Used, Total: vmStat.Total},
		Disk:      diskStats,
		Net:       netStat,
		Processes: topProcs,
	}, nil
}

// Run starts the collect-and-send loop with exponential backoff reconnect.
func (c *Collector) Run() {
	backoff := time.Second
	maxBackoff := 60 * time.Second

	for {
		err := c.connect()
		if err != nil {
			log.Printf("connection failed: %v — retrying in %v", err, backoff)
			time.Sleep(backoff)
			backoff = time.Duration(math.Min(float64(backoff*2), float64(maxBackoff)))
			continue
		}
		backoff = time.Second // reset on successful connection
	}
}

func (c *Collector) connect() error {
	u, err := url.Parse(c.serverURL)
	if err != nil {
		return err
	}
	u.Path = "/ws/agent"
	q := u.Query()
	q.Set("token", c.token)
	u.RawQuery = q.Encode()

	log.Printf("connecting to %s", u.String())
	conn, _, err := websocket.DefaultDialer.Dial(u.String(), nil)
	if err != nil {
		return err
	}
	defer conn.Close()
	log.Printf("connected as %s", c.name)

	ticker := time.NewTicker(collectInterval)
	defer ticker.Stop()
	for range ticker.C {
		payload, err := c.Collect()
		if err != nil {
			log.Printf("collect error: %v", err)
			continue
		}
		b, _ := json.Marshal(payload)
		if err := conn.WriteMessage(websocket.TextMessage, b); err != nil {
			return err
		}
	}
	return nil
}
```

```typescript
export type WSMessage = {
  type: string
  payload?: unknown
}

type Handler = (msg: WSMessage) => void

class DashboardWS {
  private ws: WebSocket | null = null
  private handlers: Handler[] = []
  private reconnectDelay = 1000
  private token: string

  constructor(token: string) {
    this.token = token
  }

  connect() {
    const protocol = window.location.protocol === "https:" ? "wss:" : "ws:"
    const url = `${protocol}//${window.location.host}/ws/dashboard?token=${this.token}`
    this.ws = new WebSocket(url)
    this.ws.onmessage = (e) => {
      try {
        const msg: WSMessage = JSON.parse(e.data)
        this.handlers.forEach((h) => h(msg))
      } catch {}
    }
    this.ws.onclose = () => {
      setTimeout(() => {
        this.reconnectDelay = Math.min(this.reconnectDelay * 2, 60000)
        this.connect()
      }, this.reconnectDelay)
    }
    this.ws.onopen = () => { this.reconnectDelay = 1000 }
  }

  on(handler: Handler) { this.handlers.push(handler) }
  off(handler: Handler) { this.handlers = this.handlers.filter((h) => h !== handler) }
}

let instance: DashboardWS | null = null
export function getDashboardWS(token: string): DashboardWS {
  if (!instance) {
    instance = new DashboardWS(token)
    instance.connect()
  }
  return instance
}
```

```tsx
function ProgressBar({ value, max, color }: { value: number; max: number; color: string }) {
```

**Step 2: Create Dashboard page**

Create `frontend/src/pages/Dashboard.tsx`:

```tsx
import { useEffect, useState } from "react"
import { api, Node } from "../lib/api"
import { getDashboardWS } from "../lib/ws"
import { NodeCard } from "../components/NodeCard"

type LiveMetrics = Record<string, { cpu: number; memUsed: number; memTotal: number }>

export function Dashboard({ onSelectNode }: { onSelectNode: (id: string) => void }) {
  const [nodes, setNodes] = useState<Node[]>([])
  const [live, setLive] = useState<LiveMetrics>({})

  useEffect(() => {
    api.nodes().then((r) => setNodes(r.nodes))

    const ws = getDashboardWS(import.meta.env.VITE_ADMIN_TOKEN ?? "")
    const handler = (msg: { type: string; payload?: unknown }) => {
      if (msg.type === "metrics") {
        const { node_id, data } = msg.payload as { node_id: string; data: { cpu: number; mem: { used: number; total: number } } }
        setLive((prev) => ({ ...prev, [node_id]: { cpu: data.cpu, memUsed: data.mem.used, memTotal: data.mem.total } }))
      }
      if (msg.type === "node_status") {
        const { node_id, online } = msg.payload as { node_id: string; online: boolean }
        setNodes((prev) => prev.map((n) => (n.id === node_id ? { ...n, online } : n)))
      }
    }
    ws.on(handler)
    return () => ws.off(handler)
  }, [])

  const onlineCount = nodes.filter((n) => n.online).length
  const avgCPU = onlineCount > 0
    ? (Object.values(live).reduce((a, b) => a + b.cpu, 0) / onlineCount).toFixed(1)
    : "—"

  return (
    <div className="space-y-6">
      {/* Summary bar */}
      <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
        {[
          { label: "Online", value: `${onlineCount} / ${nodes.length}` },
          { label: "Avg CPU", value: `${avgCPU}%` },
        ].map((s) => (
          <div key={s.label} className="bg-white/[0.03] border border-white/10 rounded-xl px-4 py-3">
            <p className="text-xs text-white/40">{s.label}</p>
            <p className="text-2xl font-semibold mt-1 tracking-tight">{s.value}</p>
          </div>
        ))}
      </div>

      {/* Node grid */}
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-4">
        {nodes.map((node) => (
          <NodeCard
            key={node.id}
            node={node}
            cpu={live[node.id]?.cpu}
            memUsed={live[node.id]?.memUsed}
            memTotal={live[node.id]?.memTotal}
            onClick={() => onSelectNode(node.id)}
          />
        ))}
      </div>
    </div>
  )
}
```

**Step 2: Create NodeDetail page**

Create `frontend/src/pages/NodeDetail.tsx`:

```tsx
import { useEffect, useState } from "react"
import { api, MetricsRow, Node } from "../lib/api"
import { MetricsChart } from "../components/MetricsChart"

const RANGES = [
  { label: "1h", seconds: 3600 },
  { label: "6h", seconds: 21600 },
  { label: "24h", seconds: 86400 },
  { label: "7d", seconds: 604800 },
]

export function NodeDetail({ nodeId }: { nodeId: string }) {
  const [metrics, setMetrics] = useState<MetricsRow[]>([])
  const [range, setRange] = useState(3600)
  const [node, setNode] = useState<Node | null>(null)

  useEffect(() => {
    api.nodes().then((r) => setNode(r.nodes.find((n) => n.id === nodeId) ?? null))
  }, [nodeId])

  useEffect(() => {
    const to = Math.floor(Date.now() / 1000)
    const from = to - range
    api.metrics(nodeId, from, to).then((r) => setMetrics(r.metrics))
  }, [nodeId, range])

  const cpu = metrics.map((m) => ({ ts: m.ts, value: m.cpu }))
  const memPct = metrics.map((m) => ({ ts: m.ts, value: m.mem_total > 0 ? (m.mem_used / m.mem_total) * 100 : 0 }))
  const netRx = metrics.map((m) => ({ ts: m.ts, value: m.net_rx / 1024 }))
  const netTx = metrics.map((m) => ({ ts: m.ts, value: m.net_tx / 1024 }))

  return (
    <div className="space-y-4 max-w-5xl">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h2 className="text-lg font-semibold">{node?.name ?? nodeId}</h2>
          <p className="text-xs text-white/40">{node?.ip} · {node?.os}/{node?.arch}</p>
        </div>
        {/* Range selector */}
        <div className="flex gap-1">
          {RANGES.map((r) => (
            <button
              key={r.label}
              onClick={() => setRange(r.seconds)}
              className={`text-xs px-3 py-1.5 rounded-lg border transition-all ${
                range === r.seconds
                  ? "bg-white/10 border-white/20 text-white"
                  : "border-white/10 text-white/40 hover:text-white/70"
              }`}
            >
              {r.label}
            </button>
          ))}
        </div>
      </div>

      {/* Charts */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        <MetricsChart data={cpu} label="CPU %" color="#10b981" />
        <MetricsChart data={memPct} label="Memory %" color="#3b82f6" />
        <MetricsChart data={netRx} label="Network RX" color="#a855f7" unit=" KB" domain={[0, "auto" as unknown as number]} />
        <MetricsChart data={netTx} label="Network TX" color="#f59e0b" unit=" KB" domain={[0, "auto" as unknown as number]} />
      </div>
    </div>
  )
}
```
